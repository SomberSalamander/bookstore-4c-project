Table Books {
  BookId integer [primary key]
  Title string
  Author string
  Publisher string
  Description string
  Price float
  Stock integer
  PublishedAt datetime
  Language string
  NumberOfPages int
  Rating float
}

Table Users {
  UserId integer [primary key]
  Name string
  Email string
  Password string
  CreatedAt datetime
}

Table Carts {
  CartId integer [primary key]
  UserId integer
  CreatedAt datetime
}

Table CartItems {
  CartId integer
  BookId integer
  Quantity integer
}

Ref: Carts.CartId < CartItems.CartId

Ref: Books.BookId < CartItems.BookId

Ref: Carts.UserId < Users.UserId
